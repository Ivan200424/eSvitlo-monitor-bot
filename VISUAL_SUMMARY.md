## ğŸ“ Usage Example

```javascript
// Define states
const wizardStates = {
  selectRegion: createStateHandler({
    onEnter: async (userId) => {
      const kb = nav.createWizardKeyboard(1, 2, regions);
      await bot.sendMessage(userId, 'Select region', kb);
    },
    onInput: async (userId, input, ctx, machine) => {
      if (input.region) {
        await machine.transition(userId, 'selectQueue', {
          region: input.region
        });
        return { handled: true };
      }
      return { handled: false };
    },
    onCancel: async (userId) => {
      await bot.sendMessage(userId, 'Setup cancelled');
    },
    onTimeout: async (userId) => {
      await bot.sendMessage(userId, 'Setup timeout');
    },
    onExit: async () => {}
  }),
  
  selectQueue: createStateHandler({
    // ... similar structure
  })
};

// Create machine
const wizard = new StateMachine('wizard', wizardStates);

// Use in bot
bot.onText(/\/start/, async (msg) => {
  await wizard.start(msg.from.id, 'selectRegion');
});

bot.on('callback_query', async (query) => {
  if (wizard.isActive(query.from.id)) {
    await wizard.handleInput(query.from.id, {
      region: parseCallbackData(query.data)
    });
  }
});
```
